//Count how many times each query string appears in the list of input strings.
#include <stdio.h>
#include <string.h>


int main() {
    int n, q;
    char strings[1000][101];
    char queries[1000][101];

    
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%s", strings[i]);
    }

       scanf("%d", &q);
    for (int i = 0; i < q; i++) {
        scanf("%s", queries[i]);
    }

    for (int i = 0; i < q; i++) {
        int count = 0;
        for (int j = 0; j < n; j++) {
            if (strcmp(queries[i], strings[j]) == 0) {
                count++;
            }
        }
        printf("%d\n", count);
    }

    return 0;
}
