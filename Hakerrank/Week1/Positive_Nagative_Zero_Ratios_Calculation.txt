#include <stdio.h>

int main() {
    int n;
    scanf("%d", &n); 

    int arr[n];
    int P = 0, N = 0, Z = 0;

   
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]); 
    }

    
    for (int i = 0; i < n; i++) {
        if (arr[i] > 0)
            P++;
        else if (arr[i] < 0)
            N++;
        else
            Z++;
    }

    
    float resP = (float)P / n;
    float resN = (float)N / n;
    float resZ = (float)Z / n;

    
    printf("%.6f\n", resP);
    printf("%.6f\n", resN);
    printf("%.6f\n", resZ);

    return 0;
}
