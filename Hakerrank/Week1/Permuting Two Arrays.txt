#include <stdio.h>

int main() {
    int q;
    scanf("%d", &q);

    while (q--) {
        int n, k;
        scanf("%d %d", &n, &k);

        int A[n], B[n];

        for (int i = 0; i < n; i++) {
            scanf("%d", &A[i]);
        }

        for (int i = 0; i < n; i++) {
            scanf("%d", &B[i]);
        }

        // Sort A in ascending order
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                if (A[j] > A[j + 1]) {
                    int temp = A[j];
                    A[j] = A[j + 1];
                    A[j + 1] = temp;
                }
            }
        }

        // Sort B in descending order
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                if (B[j] < B[j + 1]) {
                    int temp = B[j];
                    B[j] = B[j + 1];
                    B[j + 1] = temp;
                }
            }
        }

        // Now check the condition A[i] + B[i] >= k
        int res = 1;
        for (int i = 0; i < n; i++) {
            if (A[i] + B[i] < k) {
                res = 0;
                break;
            }
        }

        if (res) {
            printf("YES\n");
        } else {
            printf("NO\n");
        }
    }

    return 0;
}
