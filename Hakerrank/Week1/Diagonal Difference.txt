#include <stdio.h>
#include <stdlib.h> 

int main() {
    int n;
    scanf("%d", &n); 

    int arr[n][n];

   
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &arr[i][j]);
        }
    }

    int F_D = 0;
    int s_D = 0;

    for (int i = 0; i < n; i++) {
        F_D += arr[i][i];         
        s_D += arr[i][n - 1 - i]; 
    }

    int sum = abs(F_D-s_D);
    printf("%d\n", sum);

    return 0;
}
