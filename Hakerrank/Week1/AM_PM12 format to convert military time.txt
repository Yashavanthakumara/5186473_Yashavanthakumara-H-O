//Given a time in -hour AM/PM format, convert it to military (24-hour) time.
#include<stdio.h>
#include<stdlib.h>


int main(){

    int hour;
    char time[11];
    scanf("%s",time);
    hour=(time[0]-'0')*10+(time[1]-'0');
    if(time[8]=='A'&&time[9]=='M')
    {
        if(hour==12)
        {
            hour==0;
        }
    }
    else if(time[8]=='P' && time[9]=='M')
    {
        if(hour !=12 )
        {
            hour +=12;
        }
    }
    printf("%02d",hour);
    for(int i=2;i<8; i++){
        printf("%c",time[i]);
    }
    printf("\n");
    return 0;
}
