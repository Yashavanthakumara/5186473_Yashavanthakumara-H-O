#include <stdio.h>

int main() {
    long arr[5];
    int i, j;
    long temp;

    
    for(i = 0; i < 5; i++) {
        scanf("%lld", &arr[i]);
    }

    
    for(i = 0; i < 4; i++) {
        for(j = i + 1; j < 5; j++) {
            if(arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    
    long minSum = 0, maxSum = 0;
    for(i = 0; i < 4; i++) {
        minSum += arr[i];
    }
    for(i = 1; i < 5; i++) {
        maxSum += arr[i];
    }


    printf("%ld %ld\n", minSum, maxSum);

    return 0;
}
