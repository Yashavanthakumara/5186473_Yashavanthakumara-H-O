#include<stdio.h>
int main(){
    int n, count = 0;
    scanf("%d", &n);
    int ar[n];

    for (int i = 0; i < n; i++) {
        scanf("%d", &ar[i]);
    }

    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (ar[i] > ar[j]) {
                int temp = ar[i];
                ar[i] = ar[j];
                ar[j] = temp;
            }
        }
    }

 
    int i = 0;
    while (i < n - 1) {
        if (ar[i] == ar[i + 1]) {
            count++;
            i += 2;
        } else {
            i++;
        }
    }

    printf("%d", count);
    return 0;
}
