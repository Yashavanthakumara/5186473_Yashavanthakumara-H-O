#include <stdio.h>
#include <stdlib.h>

int cmpfunc(const void *a, const void *b) {
    return (*(char *)a - *(char *)b);
}

int main() {
    int n;
    scanf("%d", &n);

    char grid[n][n + 1]; // +1 for null terminator

    for (int i = 0; i < n; i++) {
        scanf("%s", grid[i]);
    }

    // Sort each row alphabetically
    for (int i = 0; i < n; i++) {
        qsort(grid[i], n, sizeof(char), cmpfunc);
    }

    // Check columns for sorted order
    int sorted = 1;
    for (int col = 0; col < n && sorted; col++) {
        for (int row = 0; row < n - 1; row++) {
            if (grid[row][col] > grid[row + 1][col]) {
                sorted = 0;
                break;
            }
        }
    }

    if (sorted)
        printf("YES\n");
    else
        printf("NO\n");

    return 0;
}
