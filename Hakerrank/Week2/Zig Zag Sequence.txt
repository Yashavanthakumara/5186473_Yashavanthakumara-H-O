#include <stdio.h>
#include <stdlib.h>

int main() {
    int T;
    if (scanf("%d", &T) != 1) return 0;

    while (T--) {
        int n;
        scanf("%d", &n);
        int a[n];
        for (int i = 0; i < n; i++) scanf("%d", &a[i]);

               for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (a[i] > a[j]) {
                    int tmp = a[i]; a[i] = a[j]; a[j] = tmp;
                }
            }
        }

        int mid = n / 2;               


        for (int i = 0; i < mid; i++) printf("%d ", a[i]);

        
        printf("%d ", a[n - 1]);


        for (int i = n - 2; i >= mid; i--) printf("%d ", a[i]);

        printf("\n");
    }

    return 0;
}
