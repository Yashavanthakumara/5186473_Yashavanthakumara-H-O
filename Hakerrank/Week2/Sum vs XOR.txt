#include <stdio.h>

int main() {
    unsigned long long n;
    if (scanf("%llu", &n) != 1) return 0;

    
    int Bits = 0;
    unsigned long long temp = n;


    while (temp > 0) {
        if ((temp & 1ULL) == 0) Bits++;
        temp >>= 1;         
    }

    unsigned long long ans = 1;
    for (int i = 0; i < Bits; i++) ans *= 2;

    printf("%llu\n", ans);
    return 0;
}
